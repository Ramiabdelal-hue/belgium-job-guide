@app.route("/admin/store//add-product", methods=["POST"]) @admin_required def add_product(store_id): try: # معالجة رفع الصورة file = request.files.get('product_image') filename = None if file and allowed_file(file.filename): filename = f"prod_{uuid.uuid4().hex}_{secure_filename(file.filename)}" file.save(os.path.join(app.config["UPLOAD_FOLDER"], filename)) # تحويل التاريخ من نص إلى كائن Date start_dt = datetime.strptime(request.form.get("offer_start"), '%Y-%m-%d').date() if request.form.get("offer_start") else None end_dt = datetime.strptime(request.form.get("offer_end"), '%Y-%m-%d').date() if request.form.get("offer_end") else None new_product = Product( name=request.form.get("product_name"), image=filename, price_before=request.form.get("price_before") or 0, price_after=request.form.get("price_after") or 0, has_offer=True if request.form.get("has_offer") else False, offer_start=start_dt, offer_end=end_dt, notes=request.form.get("notes"), store_id=store_id ) db.session.add(new_product) db.session.commit() flash("✅ تم إضافة الصنف بنجاح") except Exception as e: db.session.rollback() flash(f"❌ حدث خطأ: {str(e)}") # بعد الإضافة نعود لصفحة الإدارة نفسها return redirect(url_for('manage_products', store_id=store_id))
